#include <stdio.h>

int main()
{
    long long int N;
    printf("Input N: \n");
    int rez = scanf("%lld", &N);
    long long int V = 0;
    int OST;
    if (rez == 1 && N < 2147483637)
    {
        while (N != 0)
        {
            OST = N % 10;
            V = V * 10 + OST;
            N = N / 10;
        }
        
            printf("%lld", V);
        
    }
    else
    {
        printf("WRONG DATA OR OVERFLOW!");
    }
   
    
    return 0;
}
