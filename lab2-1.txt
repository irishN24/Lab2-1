#include <stdio.h>
int main()
{
    int N;
    int rez = scanf("%d", &N);
    int V = 0;
    int OST = 0;
    if (rez == 1)
    {
        if (N < 10 || N > 214747412)
        {
            printf("Overflow");
        }
        else{
        while (N != 0)
        {
            OST = N % 10;
            V = V * 10 + OST;
            N = N / 10;
        }
        printf("%d", V);
        }
        
    }
    else
    {
        printf("WRONG DATA");
    }
   
    
    return 0;
}
